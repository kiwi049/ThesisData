50   def generate_stat(df: pd.DataFrame, instrument):
51       df_stat = df[['%Chg', 'Volume']].quantile([.01, .25, .50, .75, .99])
52       df1 = pd.DataFrame([[df['%Chg'].mean(), df['Volume'].mean()], [df['%Chg'].min(), df['Volume'].min()]],
53                          columns=['%Chg', 'Volume'],
54                          index=['Mean', 'Min'],
55                          )
56       df_stat = pd.concat([df1, df_stat])
57       df_2 = pd.DataFrame([[df['%Chg'].max(), df['Volume'].max()]], columns=['%Chg', 'Volume'], index=['Max'])
58       df_stat = pd.concat([df_stat, df_2])
59       df_stat['%Chg'] = df_stat['%Chg'] * 100
60       df_stat['Volume'] = df_stat['Volume'] / 1000
61       df_stat = df_stat.transpose()
62       df_stat.columns = ['Mean', 'Min', '1%', '25%', '50%', '75%', '99%', 'Max']
63       df_stat.index = ['Return(percent)', 'Volume(1,000 shares)']
64   
65       df_stat.to_excel(os.path.join('output', instrument + '_stat.xlsx'))
66       # raise Exception
67       print(f'Finished generating volatility and volume statistics')
68       return

84   def generate_plotting(df: pd.DataFrame, instrument):
85       # daily log realized vola (sum of squared one min returns)
86       df['Log Realized Vola'] = df['%Chg'] ** 2
87       df['Date String'] = df['Local Time'].dt.strftime('%Y-%m-%d')
88       df_day = df.groupby(['Date String'])[['Log Realized Vola', 'Volume']].sum()
89       df_day['Log Realized Vola'] = np.log10(((df_day['Log Realized Vola'] + 1) ** 365 - 1) * 100)
90       df_day['Volume'] = np.log10(df_day['Volume'] / 1000000)
91       df_day = df_day.reset_index()
92   
93       # create breaks to skip time gap on graphs
94       dt_all = pd.date_range(start=datetime.strptime(df_day.loc[0, 'Date String'], '%Y-%m-%d'),
95                              end=datetime.strptime(df_day.loc[len(df_day) - 1, 'Date String'], '%Y-%m-%d'))
96       dt_obs = df['Date String'].tolist()
97       dt_breaks = [d for d in dt_all.strftime('%Y-%m-%d').tolist() if d not in dt_obs]
98       dt_breaks = pd.to_datetime(dt_breaks)
99   
100      # plotting
101      fig1 = go.Figure(data=[go.Scatter(
102          x=df_day['Date String'], y=df_day['Log Realized Vola'], mode='lines'
103      )])
104      fig1.update_xaxes(rangebreaks=[dict(dvalue=24 * 60 * 60 * 1000, values=dt_breaks)])
105      fig2 = go.Figure(data=[go.Scatter(
106          x=df_day['Date String'], y=df_day['Volume'], mode='lines'
107      )])
108      fig2.update_xaxes(rangebreaks=[dict(dvalue=24 * 60 * 60 * 1000, values=dt_breaks)])
109  
110      plotly.offline.plot(fig1, filename=os.path.join('output', instrument + '_vola_plot.html'))
111      plotly.offline.plot(fig2, filename=os.path.join('output', instrument + '_volume_plot.html'))
112  
113      print(f'Finished plotting volume and volatility for {instrument}')
114      return

150  def generate_intraday_pattern(df, df_ann, instrument):
151      df['Date String'] = df['Local Time'].dt.strftime('%Y-%m-%d')
152      df_ann['Date String'] = df_ann['Date Time'].dt.strftime('%Y-%m-%d')
153      df_intraday = df[~df['Date String'].isin(df_ann['Date String'])].copy()
154      df_intraday['Minute'] = df['Local Time'].dt.strftime('%H:%M')
155      df_intraday['Intra Vola'] = np.sqrt(df_intraday['%Chg'].pow(2))
156      df_intraday = df_intraday.groupby('Minute')[['Intra Vola', 'Volume']].mean()
157      df_intraday = df_intraday.dropna()
158      df_intraday['Intra Vola'] = ((df_intraday['Intra Vola'] + 1).pow(365) - 1) * 100
159      df_intraday = df_intraday.reset_index()
160  
161      df_intraday_vola = df_intraday[['Minute', 'Intra Vola']]
162      df_intraday_vola = df_intraday_vola[(np.abs(stats.zscore(df_intraday_vola['Intra Vola'])) < 1.2)]
163      df_intraday_vola['Smooth'] = np.polyval(np.polyfit(df_intraday_vola.index, df_intraday_vola['Intra Vola'], 10),
164                                              df_intraday_vola.index)
165  
166      df_intraday_volume = df_intraday[['Minute', 'Volume']]
167      df_intraday_volume = df_intraday_volume[(np.abs(stats.zscore(df_intraday_volume['Volume'])) < 1.2)]
168      df_intraday_volume['Smooth'] = np.polyval(np.polyfit(df_intraday_volume.index, df_intraday_volume['Volume'], 10),
169                                                df_intraday_volume.index)
170  
171      fig1 = px.scatter(df_intraday_vola, x='Minute', y='Intra Vola')
172      fig1.add_trace(go.Scatter(x=df_intraday_vola['Minute'], y=df_intraday_vola['Smooth'], name=None), secondary_y=False)
173      fig2 = px.scatter(df_intraday_volume, x='Minute', y='Volume')
174      fig2.add_trace(go.Scatter(x=df_intraday_volume['Minute'], y=df_intraday_volume['Smooth'], name=None),
175                     secondary_y=False)
176  
177      plotly.offline.plot(fig1, filename=os.path.join('output', instrument + '_intraday_vola.html'))
178      plotly.offline.plot(fig2, filename=os.path.join('output', instrument + '_intraday_volume.html'))
179  
180      print(f'Finished generating intraday pattern on non-announcement days for {instrument}')
181      return

184  def generate_ann_pattern(df, df_news, instrument, war_flag=False):
185      if war_flag:
186          region_list = [['EC']]
187      elif instrument == 'stoxx':
188          region_list = [['EC'], ['US'], ['EC', 'US']]
189      else:
190          region_list = [['GE'], ['EC', 'US', 'GE']]
191  
192      for region in region_list:
193          region_str = '_'.join(region)
194          df_news_region = df_news[df_news['Region'].isin(region)]
195          pattern_vola_data = []
196          pattern_volume_data = []
197          for i, row in df_news_region.iterrows():
198              min_range = list(range(-15, 16))
199              news_vola_line = []
200              news_volume_line = []
201              for minute in min_range:
202                  dt = row['Date Time'] + timedelta(minutes=minute)
203                  try:
204                      news_vola_line.append(df.loc[df['Local Time'] == dt, '%Chg'].iloc[0])
205                      news_volume_line.append(df.loc[df['Local Time'] == dt, 'Volume'].iloc[0])
206                  except IndexError:
207                      news_vola_line.append(np.nan)
208                      news_volume_line.append(np.nan)
209  
210              pattern_vola_data.append(news_vola_line)
211              pattern_volume_data.append(news_volume_line)
212          columns = list(range(-15, 16))
213  
214          df_pattern_vola = pd.DataFrame(pattern_vola_data, columns=columns)
215          df_pattern_vola = df_pattern_vola.dropna(how='all').pow(2)
216          df_pattern_volume = pd.DataFrame(pattern_volume_data, columns=columns)
217          df_pattern_volume = df_pattern_volume.dropna(how='all')
218  
219          pattern_vola = df_pattern_vola.mean()
220          pattern_vola = pattern_vola.rename('Volatility')
221          pattern_vola = ((np.sqrt(pattern_vola) + 1) ** 365 - 1) * 100
222          pattern_volume = df_pattern_volume.mean() / 100000
223          pattern_volume = pattern_volume.rename('Volume')
224  
225          fig1 = px.scatter(pattern_vola, labels=dict(index='Time', value='Spot Volatility'))
226          fig1.update_traces(marker={'size': 15})
227          fig1.add_vline(x=0, line_dash="dash")
228          fig2 = px.scatter(pattern_volume, labels=dict(index='Time', value='Volume Intensity'))
229          fig2.update_traces(marker={'size': 15})
230          fig2.add_vline(x=0, line_dash="dash")
231  
232          name = instrument + '_war_' + region_str if war_flag else instrument + '_' + region_str
233          plotly.offline.plot(fig1, filename=os.path.join('output', name + '_news_vola.html'))
234          plotly.offline.plot(fig2, filename=os.path.join('output', name + '_news_volume.html'))
235  
236      if war_flag:
237          print(f'Plotted the pattern before and after the war announcements for {instrument}')
238      else:
239          print(f'Plotted the pattern before and after the news announcements for {instrument}')
240      return

318  
319          if df_pre.empty or df_post.empty or series_event.empty:
320              continue
321          else:
322              line_data = [event_dt, series_event[0],
323                           np.sqrt(df_pre['Volatility'].mean()),
324                           df_pre['Volume'].mean() / 10000,
325                           np.sqrt(df_post['Volatility'].mean()),

329      columns = ['Date Time', 'Minute Return', 'Pre Vola', 'Pre Volume', 'Post Vola', 'Post Volume']

338      fig.add_trace(trace_pre)
339      fig.add_trace(trace_post, secondary_y=False)
340      fig.update_xaxes(title_text='Normalized Return')
341      fig.update_yaxes(title_text='Average Volume     10,000 shares')
342      #fig.update_xaxes(tickvals=df_norm.index, ticktext=df_norm['Norm Return'], tickmode='array')
343  
344      name = instrument + '_war_' if war_flag else instrument
345      plotly.offline.plot(fig, filename=os.path.join('output', name + '_normalized_return.html'))

353  def generate_jump_regression(df, df_news, instrument, war_flag=False):
354      if war_flag:
355          region = ['EC']
356      elif instrument == 'stoxx':
357          region = ['EC', 'US']
358      else:
359          region = ['EC', 'US', 'GE']
360      df_news_region = df_news[df_news['Region'].isin(region)]
361      df_jump_list = []
362      for i, row in df_news_region.iterrows():
363          min_range_pre = list(range(-14, 1))
364          min_range_post = list(range(2, 17))
365          pre_dt_list = [row['Date Time'] + timedelta(minutes=minute) for minute in min_range_pre]
366          post_dt_list = [row['Date Time'] + timedelta(minutes=minute) for minute in min_range_post]
367          df_pre = df[df['Local Time'].isin(pre_dt_list)].copy()
368          df_post = df[df['Local Time'].isin(post_dt_list)].copy()
369  
370          if df_pre.empty or df_post.empty:
371              continue
372          else:
373              line_data = [np.log10(np.sqrt(df_post['%Chg'].pow(2).mean())) - np.log10(np.sqrt(df_pre['%Chg'].pow(2).mean())),
374                           np.log10(df_post['Volume'].mean()) - np.log10(df_pre['Volume'].mean())]
375              df_jump_list.append(line_data)
376      columns = ['Log Volatility Jump', 'Log Volume Intensity Jump']
377      df_jump = pd.DataFrame(df_jump_list, columns=columns)
378      df_jump = df_jump[(df_jump['Log Volatility Jump'] > 0) & (df_jump['Log Volume Intensity Jump'] > 0)]
379  
380      fig = px.scatter(df_jump, x='Log Volatility Jump', y='Log Volume Intensity Jump', trendline='ols')
381      name = instrument + '_war_' if war_flag else instrument
382      plotly.offline.plot(fig, filename=os.path.join('output', name + '_vola_jump.html'))
383      if war_flag:
384          print(f'Plotted vola jump of war events for {instrument}')
385          print(f'Average log volatility jump of war events for {instrument}: {df_jump["Log Volatility Jump"].mean()}')
386          print(f'Average log volume intensity jump of war events for {instrument}: {df_jump["Log Volume Intensity Jump"].mean()}')
387      else:
388          print(f'Plotted vola jump of announcements for {instrument}')
389          print(f'Average log volatility jump of announcements for {instrument}: {df_jump["Log Volatility Jump"].mean()}')
390          print(f'Average log volume intensity jump of announcements for {instrument}: {df_jump["Log Volume Intensity Jump"].mean()}')
391      results = px.get_trendline_results(fig)
392      print(results.px_fit_results.iloc[0].summary())
393  
394      return

401          region = ['EC', 'US', 'GE']
402      df_news_region = df_news[df_news['Region'].isin(region)]
403      event_list = df_news_region['Event'].unique()
404      event_list = np.insert(event_list, 0, 'All')
405      stat_data = []
406      for event in event_list:
407          if event == 'All':
408              df_event = df_news
409          else:
410              df_event = df_news[df_news['Event'] == event].reset_index(drop=True)
411          df_jump_list = []
412          for i, row in df_event.iterrows():
413              min_range_pre = list(range(-14, 1))
414              min_range_post = list(range(2, 17))
415              pre_dt_list = [row['Date Time'] + timedelta(minutes=minute) for minute in min_range_pre]
416              post_dt_list = [row['Date Time'] + timedelta(minutes=minute) for minute in min_range_post]
417              df_pre = df[df['Local Time'].isin(pre_dt_list)].copy()
418              df_post = df[df['Local Time'].isin(post_dt_list)].copy()
419  
420              if df_pre.empty or df_post.empty:
421                  continue
422              else:
423                  line_data = [
424                      np.log10(np.sqrt(df_post['%Chg'].pow(2).mean())) - np.log10(np.sqrt(df_pre['%Chg'].pow(2).mean())),

426                  df_jump_list.append(line_data)
427          columns = ['Log Volatility Jump', 'Log Volume Intensity Jump']

440      df_stat.to_excel(os.path.join('output', instrument + '_announcements_stat.xlsx'), index=False)
